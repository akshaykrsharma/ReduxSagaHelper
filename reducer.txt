import { KEY, KEY_SUCCESS, KEY_ERROR } from '../type';

const INITIAL_STATE = {
	error: '',
	isFetching: true,
	CAMELKEYData: {}
};

const CAMELKEYReducer = (state = INITIAL_STATE, action: reduxPropType) => {
	switch (action.type) {
		case KEY:
			return {
				...state,
				isFetching: true,
				payload: null
			};
		case KEY_SUCCESS:
			return {
				...state,
				payload: action.payload,
				isFetching: false,
				error: null
			};
		case KEY_ERROR:
			return {
				...state,
				isFetching: false,
				error: action.payload
			};

		default:
			return state ? {} : INITIAL_STATE;
	}
};

module.exports = {
	CAMELKEYReducer
};
